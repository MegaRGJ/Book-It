## Create the BookIt datbase with all tables
CREATE DATABASE BookIT
CREATE TABLE Users(
UserID int NOT NULL PRIMARY KEY,
Name varchar(255),
Address varchar(255),
Postcode varchar(255),
Email varchar(255),
Phone varchar(255),
Username varchar(255),
Password varchar(255),
LastLogin varchar(255),
Salt varchar(255),
Permission  varchar(255),
)
CREATE TABLE Businesses(
BusinessID int NOT NULL PRIMARY KEY,
Name varchar(255),
Address varchar(255),
Postcode varchar(255),
Email varchar(255),
Phone varchar(255),
Description  varchar(255),
Active varchar(255),
SharedBookings varchar(255),
)
CREATE TABLE Employees(
EmployeeID int NOT NULL PRIMARY KEY,
BusinessID int NOT NULL,
UserID int NOT NULL,
PermissionLevel varchar(255),
Availability varchar(255),
Ammendments varchar(255),
)
CREATE TABLE Bookings(
BookingID int NOT NULL PRIMARY KEY,
EmployeeID   int NOT NULL,
UserID int NOT NULL,
DateTime varchar(255),
Duration varchar(255),
Description varchar(255),
Cancellation varchar(255),
)
CREATE TABLE MailBox(
MailID int NOT NULL PRIMARY KEY,
UserID int NOT NULL,
EmployeeID int NOT NULL,
Message varchar(255),
DateTime varchar(255),
)
CREATE TABLE Invites(
InviteID int NOT NULL PRIMARY KEY,
BusinessID int NOT NULL,
UserID int NOT NULL,
InviteCode varchar(255),
DateTime varchar(255),
Expires varchar(255),
Uses varchar(255),
)
